    def to_s
      if(@VAR_OBJECT == nil)
        return "ID=[" + @VAR_ID.to_s() + "], Name=[" + @VAR_NAME + "], Value=[nil]"
      else
        return "ID=[" + @VAR_ID.to_s() + "], Name=[" + @VAR_NAME + "], Value=[" + @VAR_OBJECT.to_s() + "], Type=[" + @VAR_OBJECT.class.to_s() + "]"
      end
    end
